702 /*
703 When defined to value other than 0, the library will try to use
704 D3D12_SMALL_RESOURCE_PLACEMENT_ALIGNMENT or D3D12_SMALL_MSAA_RESOURCE_PLACEMENT_ALIGNMENT
705 for created textures when possible, which can save memory because some small textures
706 may get their alignment 4K and their size a multiply of 4K instead of 64K.
707 
708 #define D3D12MA_USE_SMALL_RESOURCE_PLACEMENT_ALIGNMENT 0
709  Disables small texture alignment.
710 #define D3D12MA_USE_SMALL_RESOURCE_PLACEMENT_ALIGNMENT 1
711  Enables conservative algorithm that will use small alignment only for some textures
712  that are surely known to support it.
713 #define D3D12MA_USE_SMALL_RESOURCE_PLACEMENT_ALIGNMENT 2
714  Enables query for small alignment to D3D12 (based on Microsoft sample) which will
715  enable small alignment for more textures, but will also generate D3D Debug Layer
716  error #721 on call to ID3D12Device::GetResourceAllocationInfo, which you should just
717  ignore.
718 */
719 #ifndef D3D12MA_USE_SMALL_RESOURCE_PLACEMENT_ALIGNMENT
720  #define D3D12MA_USE_SMALL_RESOURCE_PLACEMENT_ALIGNMENT 1
721 #endif
